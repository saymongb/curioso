SELECT U.SQL_FULLTEXT
  FROM V$SESSION V, V$SQL U
 WHERE V.USERNAME LIKE 'LUKAS_RORIZ'
   AND (V.SQL_ID = U.sql_id OR V.PREV_SQL_ID = U.SQL_ID)
   AND UPPER(V.MODULE) LIKE '%CESGESTOR.EXE%'
   AND V.STATUS = 'ACTIVE'
   AND UPPER(V.TERMINAL) LIKE 'SUPORTE4'
   AND UPPER(V.OSUSER) LIKE 'USER'
   AND USERNAME NOT IN
       ('SAC_CIS', 'SAC_SUPORTE', 'JUMANJI', 'CIS_BH', 'ZANK')
 ORDER BY U.LAST_ACTIVE_TIME
