--SELECT SYS_CONTEXT('USERENV','TERMINAL'),SYS_CONTEXT('USERENV','HOST')  FROM --dual;

SELECT UPPER(V.PROGRAM)
  FROM V$SESSION V
 WHERE UPPER(V.USERNAME) = 'FABIO_FORT'
   AND UPPER(V.TERMINAL) LIKE '%SUPORTE4'
   AND UPPER(V.OSUSER) LIKE '%CIS'
   AND UPPER(USERNAME) NOT IN
       ('SAC_CIS', 'SAC_SUPORTE', 'JUMANJI', 'CIS_BH', 'ZANK')
